1: """
2: vtuber_bridge.py - Open-LLM-VTuber 桥接模块
3: 
4: 通过 WebSocket 连接到 Open-LLM-VTuber 后端 (ws://127.0.0.1:12393/client-ws)
5: 发送 text-input 消息触发 AI 对话。
6: 
7: 后端默认端口: 12393
8: """
9: import json
10: import logging
11: import threading
12: import time
13: import asyncio
14: 
15: log = logging.getLogger("vtuber_bridge")
16: 
17: # websocket-client 在 requirements.txt 中已声明
18: try:
19:     import websocket
20:     HAS_WS = True
21: except ImportError:
22:     HAS_WS = False
23: 
24: 
25: class VTuberBridge:
26:     """
27:     通过 WebSocket 连接到 Open-LLM-VTuber 的桥接器。
28: 
29:     Open-LLM-VTuber 后端端口: 12393 (conf.yaml 中配置)
30:     WebSocket 端点: ws://<host>:<port>/client-ws
31:     """
32: 
33:     def __init__(
34:         self,
35:         backend_url: str = "http://127.0.0.1:12393",
36:         enabled: bool = False,
37:     ):
38:         # 存储 HTTP URL，展示给用户，连接时转换为 WS
39:         self._http_url = backend_url.rstrip("/")
40:         self.backend_url = backend_url
41:         self.enabled = enabled
42:         self._ws = None
43:         self._thread: threading.Thread | None = None
44:         self._lock = threading.Lock()
45:         self._running = False
46:         self._retry_count = 0
47:         self._max_retries = 3
48: 
49:     def _connect(self) -> bool:
50:         """在新线程中建立 WebSocket 连接"""
51:         if not HAS_WS:
52:             log.warning("[VTuberBridge] websocket-client 未安装")
53:             return False
54:         # HTTP URL -> WS URL
55:         ws_url = self._http_url.replace("http://", "ws://").replace("https://", "wss://")
56:         ws_url = f"{ws_url}/client-ws"
57:         try:
58:             ws = websocket.WebSocketApp(
59:                 ws_url,
60:                 on_message=self._on_message,
61:                 on_error=self._on_error,
62:                 on_close=self._on_close,
63:                 on_open=self._on_open,
64:             )
65:             self._ws = ws
66:             self._running = True
67:             self._thread = threading.Thread(target=ws.run_forever, daemon=True)
68:             self._thread.start()
69:             log.info(f"[VTuberBridge] WebSocket 连接线程已启动: {self.backend_url}")
70:             return True
71:         except Exception as e:
72:             log.warning(f"[VTuberBridge] WebSocket 连接失败: {e}")
73:             return False
74: 
75:     def _disconnect(self):
76:         """断开 WebSocket 连接"""
77:         with self._lock:
78:             self._running = False
79:             if self._ws:
80:                 try:
81:                     self._ws.close()
82:                 except Exception:
83:                     pass
84:                 self._ws = None
85: 
86:     def _on_message(self, ws, message: str):
87:         """收到服务端消息（通常不需要处理）"""
88:         try:
89:             data = json.loads(message)
90:             log.debug(f"[VTuberBridge] 收到消息: {data.get('type', message[:100])}")
91:         except Exception:
92:             pass
93: 
94:     def _on_error(self, ws, error):
95:         log.warning(f"[VTuberBridge] WebSocket 错误: {error}")
96:         self._retry_count += 1
97: 
98:     def _on_close(self, ws, close_status_code, close_msg):
99:         log.info(f"[VTuberBridge] WebSocket 关闭: {close_status_code} {close_msg}")
100:         self._running = False
101: 
102:     def _on_open(self, ws):
103:         log.info("[VTuberBridge] WebSocket 已连接")
104:         self._retry_count = 0
105: 
106:     def _send(self, data: dict) -> bool:
107:         """发送 JSON 消息到 WebSocket"""
108:         if not self._running or not self._ws:
109:             return False
110:         try:
111:             with self._lock:
112:                 self._ws.send(json.dumps(data, ensure_ascii=False))
113:             return True
114:         except Exception as e:
115:             log.debug(f"[VTuberBridge] 发送失败: {e}")
116:             return False
117: 
118:     def _ensure_connected(self) -> bool:
119:         """确保连接活跃，断了则重连"""
120:         if not self.enabled:
121:             return False
122:         with self._lock:
123:             if self._running and self._ws:
124:                 return True
125:         # 需要重连
126:         if self._retry_count < self._max_retries:
127:             return self._connect()
128:         return False
129: 
130:     def notify_event(self, message: str) -> bool:
131:         """
132:         向桌宠推送感知事件，触发 AI 回复 + TTS + 动画。
133:         """
134:         if not self._ensure_connected():
135:             return False
136:         ok = self._send({
137:             "type": "text-input",
138:             "text": message,
139:         })
140:         if ok:
141:             log.debug(f"[VTuberBridge] 已发送: {message[:50]}")
142:         return ok
143: 
144:     def speak(self, text: str) -> bool:
145:         """直接让桌宠说一段话"""
146:         if not self._ensure_connected():
147:             return False
148:         return self._send({
149:             "type": "ai-speak-signal",
150:             "text": text,
151:         })
152: 
153:     def set_expression(self, expression: str) -> bool:
154:         """设置桌宠表情"""
155:         if not self._ensure_connected():
156:             return False
157:         return self._send({
158:             "type": "expression-change",
159:             "expression": expression,
160:         })
161: 
162:     def play_motion(self, motion: str) -> bool:
163:         """播放指定动作"""
164:         if not self._ensure_connected():
165:             return False
166:         return self._send({
167:             "type": "motion-change",
168:             "motion": motion,
169:         })
170: 
171:     def close(self):
172:         """关闭桥接，释放资源"""
173:         self._disconnect()
